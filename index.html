No mutable variables
No
var someVariable = "default value"

if (someCondition) {
    someVariable = "another value"
}
Yes
val someVariable = if (someCondition) {
    "another value"
} else {
    "default value"
}
No mutable class fields
No
data class SomeClass(
    var someField: String)

val someObject = SomeClass(someField = "foo")
someObject.someField = "bar
Yes
data class SomeClass(
    val someField: String)

val someObject = SomeClass(someField = "foo")
val someObject2 = someObject.copy(someField = "bar")
Immutable platform types
No
import java.util.Calendar
import java.util.Date
import java.util.GregorianCalendar
import java.net.URL

val someDate: Date = GregorianCalendar(2012, Calendar.DECEMBER, 4).time

val someUrl = URL("https://github.com")
Yes
import java.time.LocalDate
import java.time.Month
import java.net.URI


val someDate: LocalDate = LocalDate.of(2012, Month.DECEMBER, 4)

val someUri = URI("https://github.com")
Collection extension functions instead of loops
No
val names = listOf("Alice", "Bob", "Cedric")
val lengths = mutableListOf()

for (name in names) {
    if (name.length > 3) {
        lengths.add(name.length)
    }
}
Yes
val names = listOf("Alice", "Bob", "Cedric")
val lengths = names
    .map { it.length }
    .filter { it > 3 }
Higher-order functions instead of annotations
No
@SomeAnnotation(someParameter = 123)

    // ...
}
Yes
fun someFunction() = someWrapperFunction(someParameter = 123) {
    // ...
}
